SELECT COUNT(*) AS COUNT
FROM EMPLOYEES
WHERE ACTIVE = TRUE;

***2
SELECT COUNT (DISTINCT JOBPOSITION) AS COUNT
FROM EMPLOYEES;

***3
SELECT FIRSTNAME,
	LASTNAME
FROM EMPLOYEES
WHERE SALARY >= 2000;

***4
SELECT JOBPOSITION
FROM EMPLOYEES
GROUP BY JOBPOSITION
HAVING AVG(SALARY) > 2000;

***5
SELECT EMPLOYEES.FIRSTNAME,
	EMPLOYEES.LASTNAME
FROM EMPLOYEES
INNER JOIN MANAGERS ON EMPLOYEES.MANAGERID = MANAGERS.MANAGERID
WHERE MANAGERS.FIRSTNAME like 'M%';
